SELECT 
    A.COMPLAINT_NO, 
    A.COMPLAINT_DATE, 
    A.VENDOR_CODE, 
    A.VENDOR_NAME, 
    A.PROBLEM_STATMT, 
    A.PROC_CAT_DESC, 
    A.COMPAINT_CAT_DESC,
    B.FBACK_JUSCO, 
    B.FBACK_JUSCODATE, 
    B.SPL_COMMENT_JUSCO, 
    B.SPL_COM_JUSCODATE, 
    B.FBACK_BY_VENDOR, 
    B.FBACK_BY_VENDDATE
FROM App_COMPLAINT_HELP_REGISTER A
LEFT JOIN App_COMPLAINT_HELP_DETAILS B ON A.ID = B.MasterID
WHERE A.VENDOR_CODE = '10104'
ORDER BY A.CREATED_ON DESC, A.COMPLAINT_NO;
